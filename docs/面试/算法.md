#### find middle

public static Node findMiddle(Node head) {
    if (head == null) {
        return null;
    }
    Node pa = head, pb = head.next.next;
    while (pa != null && pb != null) {
        pa = pa.next;
        if (pb.next != null) {
            pb = pb.next.next;
        } else {
            break;
        }
    }
    return pa;
}
